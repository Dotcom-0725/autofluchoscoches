   AUTO FLUCHOS COCHES — Booking engine + Admin Dashboard backend
   (Google Apps Script Web App, backed by a Google Sheet)

   WHAT THIS DOES (v6 — adds sold-out availability notifications)
   - Everything from v5 (payment system: payment_method/payment_status/
     booking_status + optional customer_email on bookings, Bank Transfer/Wise

     "Notify me" panel (see `requestAvailabilityNotification` below), stored
     in a new "AvailabilityRequests" sheet. The existing hourly
     autoExpireBookings trigger now also calls `notifyAvailabilityRequests_`
     right after freeing expired bookings, since that's exactly the moment a
     sold-out car might have stock again — it emails any pending request
     whose contact looks like an email address, then clears it.
   - Also exposes each car's active bookings' return dates (`bookedReturns`)
     from the public (no-action) endpoint, so the site can tell a customer
     the specific date a sold-out car frees up, not just that it's sold out.
   - Everything from v4 (admin dashboard data layer: Cars/Settings/ActivityLog
     sheets, single shared admin password, stock-aware inventory, overbooking
     prevention, auto-expiry, seasonal pricing, automatic confirmation email).
   - Fleet data (the car catalog) lives in a "Cars" sheet tab instead of being
     hardcoded in index.html, so the admin dashboard (admin.html) can actually
     create/edit/delete/duplicate vehicles.
   - "Settings" sheet stores SEO meta title/description overrides that
     index.html applies at runtime.
   - "ActivityLog" sheet records every mutating admin action (who did
     what, when) for the dashboard's Activity Logs screen.

   HOW TO DEPLOY / UPDATE:
   1. Open the "Auto Fluchos Coches - Bookings" spreadsheet.
   2. Extensions > Apps Script.
   3. Replace Code.gs with the ENTIRE contents of this file.
   4. Change the passwords in ADMIN_ROLES below if you want different ones.
   5. Save (Ctrl+S).
   6. Deploy > Manage deployments > edit (pencil) > Version: "New version" > Deploy.

const AVAILABILITY_SHEET_NAME = 'AvailabilityRequests';

// Manager/Employee tiers. Change it here any time; redeploy afterwards.
const ADMIN_ROLES = {
  'Nadir2026': 'super_admin',
};
const EDITOR_ROLES = ['super_admin', 'manager']; // allowed to edit fleet/pricing/settings

const NOTIFICATION_EMAIL = 'Contact@autofluchoscoches.com';
const SITE_URL = 'https://autofluchoscoches.com/';

function roleForPassword_(password) {
  return ADMIN_ROLES[password] || null;
}

function doGet(e) {
  const action = e.parameter.action;

  if (action === 'admin') {
    const role = roleForPassword_(e.parameter.password);
    if (!role) {
      return jsonResponse_({ success: false, error: 'Wrong password' });
    }
    return jsonResponse_({
      success: true,
      role: role,
      bookings: listActiveBookings_(),
      seasons: listSeasons_(),
      cars: listCars_(),
      settings: listSettings_(),
      activityLog: listActivityLog_(),
    });
  }

  if (action === 'seasons') {
    return jsonResponse_({ seasons: listSeasons_() });
  }

  if (action === 'cars') {
    // Public: the live site's fleet section reads the car catalog from here.
    return jsonResponse_({ cars: listCars_() });
  }

  if (action === 'settings') {
    // Public: index.html applies SEO meta overrides (title/description) from here.
    return jsonResponse_({ settings: listSettings_() });
  }

  // Public: for each car, how many units are currently booked overall (kept
  // for backward compatibility), every active booking's return date
  // (bookedReturns — the Nth soonest return date is the moment the (N+1)th
  // unit becomes free, used by the "notify me" feature), AND the full
  // pickup/return interval of each active booking (bookedIntervals) so the
  // site can tell whether a car is free for a SPECIFIC requested date range
  // rather than just "is it booked right now overall" — a car with a
  // booking next week must still be bookable for a period that doesn't
  // overlap it.
  const bookedCounts = {};
  const bookedReturns = {};
  const bookedIntervals = {};
  listActiveBookings_().forEach(b => {
    bookedCounts[b.carId] = (bookedCounts[b.carId] || 0) + 1;
    if (!bookedReturns[b.carId]) bookedReturns[b.carId] = [];
    if (b.ret) bookedReturns[b.carId].push(new Date(b.ret).toISOString());
    if (!bookedIntervals[b.carId]) bookedIntervals[b.carId] = [];
    if (b.pickup && b.ret) {
      bookedIntervals[b.carId].push({
        pickup: new Date(b.pickup).toISOString(),
        ret: new Date(b.ret).toISOString(),
      });
    }
  });
  Object.keys(bookedReturns).forEach(carId => bookedReturns[carId].sort());
  return jsonResponse_({ bookedCounts: bookedCounts, bookedReturns: bookedReturns, bookedIntervals: bookedIntervals });
}

/* True if [startA, endA) overlaps [startB, endB) — the standard interval
   overlap test. Two bookings for the same car conflict only if their date
   ranges actually overlap; a booking ending July 25 does not conflict with
   one starting July 25 or later. */
function rangesOverlap_(startA, endA, startB, endB) {
  return new Date(startA) < new Date(endB) && new Date(startB) < new Date(endA);
}

function doPost(e) {
  const body = JSON.parse(e.postData.contents);

  if (body.action === 'cancel') {
    const role = roleForPassword_(body.password);
    if (!role) return jsonResponse_({ success: false, error: 'Wrong password' });
    if (!body.row) return jsonResponse_({ success: false, error: 'Missing row' });
    getBookingsSheet_().deleteRow(Number(body.row));
    logActivity_(role, 'cancel_booking', 'Row ' + body.row);
    return jsonResponse_({ success: true });
  }

  if (body.action === 'addSeason') {
    const role = roleForPassword_(body.password);
    if (!role || EDITOR_ROLES.indexOf(role) === -1) return jsonResponse_({ success: false, error: 'Not allowed' });
    if (!body.name || !body.start || !body.end || !body.value) {
      return jsonResponse_({ success: false, error: 'Missing fields' });
    }
    getSeasonsSheet_().appendRow([
      body.name, body.start, body.end, body.type === 'fixed' ? 'fixed' : 'percent', Number(body.value),
    ]);
    logActivity_(role, 'add_season', body.name);
    return jsonResponse_({ success: true });
  }

  if (body.action === 'deleteSeason') {
    const role = roleForPassword_(body.password);
    if (!role || EDITOR_ROLES.indexOf(role) === -1) return jsonResponse_({ success: false, error: 'Not allowed' });
    if (!body.row) return jsonResponse_({ success: false, error: 'Missing row' });
    getSeasonsSheet_().deleteRow(Number(body.row));
    logActivity_(role, 'delete_season', 'Row ' + body.row);
    return jsonResponse_({ success: true });
  }

  if (body.action === 'saveCar') {
    const role = roleForPassword_(body.password);
    if (!role || EDITOR_ROLES.indexOf(role) === -1) return jsonResponse_({ success: false, error: 'Not allowed' });
    const car = body.car || {};
    if (!car.id || !car.name) return jsonResponse_({ success: false, error: 'Missing id/name' });
    upsertCar_(car);
    logActivity_(role, 'save_car', car.id + ' — ' + car.name);
    return jsonResponse_({ success: true });
  }

  if (body.action === 'deleteCar') {
    const role = roleForPassword_(body.password);
    if (!role || EDITOR_ROLES.indexOf(role) === -1) return jsonResponse_({ success: false, error: 'Not allowed' });
    if (!body.id) return jsonResponse_({ success: false, error: 'Missing id' });
    deleteCar_(body.id);
    logActivity_(role, 'delete_car', body.id);
    return jsonResponse_({ success: true });
  }

  if (body.action === 'duplicateCar') {
    const role = roleForPassword_(body.password);
    if (!role || EDITOR_ROLES.indexOf(role) === -1) return jsonResponse_({ success: false, error: 'Not allowed' });
    if (!body.id) return jsonResponse_({ success: false, error: 'Missing id' });
    const newId = duplicateCar_(body.id);
    if (!newId) return jsonResponse_({ success: false, error: 'Car not found' });
    logActivity_(role, 'duplicate_car', body.id + ' -> ' + newId);
    return jsonResponse_({ success: true, id: newId });
  }

  if (body.action === 'saveSetting') {
    const role = roleForPassword_(body.password);
    if (!role || EDITOR_ROLES.indexOf(role) === -1) return jsonResponse_({ success: false, error: 'Not allowed' });
    if (!body.key) return jsonResponse_({ success: false, error: 'Missing key' });
    setSetting_(body.key, body.value || '');
    logActivity_(role, 'save_setting', body.key);
    return jsonResponse_({ success: true });
  }

  // Customer-initiated: called right after they pick a payment method in the
  // payment step (Bank Transfer "I've paid" click, or Wise "Pay with Wise"
  // click). No password — the customer isn't authenticated at this point.
  if (body.action === 'setBookingPayment') {
    if (!body.ref) return jsonResponse_({ success: false, error: 'Missing ref' });
    const row = findBookingRowByRef_(body.ref);
    if (row === -1) return jsonResponse_({ success: false, error: 'Booking not found' });
    const sheet = getBookingsSheet_();
    sheet.getRange(row, 13, 1, 3).setValues([[
      body.paymentMethod || 'none',
      body.paymentStatus || 'pending',
      body.bookingStatus || 'pending_payment',
    ]]);
    return jsonResponse_({ success: true });
  }

  // Admin-only: change a booking's payment status from the dashboard
  // (Pending / Awaiting Verification / Paid / Cancelled / Refunded).
  // Marking it Paid auto-confirms the booking and emails the customer.
  if (body.action === 'adminSetPaymentStatus') {
    const role = roleForPassword_(body.password);
    if (!role) return jsonResponse_({ success: false, error: 'Wrong password' });
    if (!body.ref || !body.paymentStatus) return jsonResponse_({ success: false, error: 'Missing fields' });
    const row = findBookingRowByRef_(body.ref);
    if (row === -1) return jsonResponse_({ success: false, error: 'Booking not found' });
    const sheet = getBookingsSheet_();
    sheet.getRange(row, 14).setValue(body.paymentStatus);
    let bookingStatus = sheet.getRange(row, 15).getValue() || 'pending_payment';
    if (body.paymentStatus === 'paid') {
      bookingStatus = 'confirmed';
      sheet.getRange(row, 15).setValue(bookingStatus);
      sendPaymentConfirmedEmail_(sheet.getRange(row, 1, 1, 15).getValues()[0]);
    } else if (body.paymentStatus === 'cancelled') {
      bookingStatus = 'cancelled';
      sheet.getRange(row, 15).setValue(bookingStatus);
    }
    logActivity_(role, 'set_payment_status', body.ref + ' — ' + body.paymentStatus);
    return jsonResponse_({ success: true });
  }

  // Customer-initiated: a visitor asked to be told when a sold-out car frees
  // up. No password — stored for autoExpireBookings (see below) to pick up
  // and auto-email once the car has free stock again.
  if (body.action === 'requestAvailabilityNotification') {
    if (!body.carId || !body.contact) return jsonResponse_({ success: false, error: 'Missing fields' });
    getAvailabilityRequestsSheet_().appendRow([
      body.carId, body.carName || '', body.contact, new Date(), '',
    ]);
    return jsonResponse_({ success: true });
  }

  // Default action: record a new booking (website checkout OR a manual
  // reservation created from the admin dashboard — same code path, so
  // inventory updates automatically either way).
  if (!body.carId) {
    return jsonResponse_({ success: false, error: 'Missing carId' });
  }
  const lock = LockService.getScriptLock();
  lock.waitLock(10000);
  try {
    const stock = Number(body.stock) || 1;
    // Only count bookings whose date range actually overlaps the one being
    // requested — a car with an existing booking that ends before this one
    // starts (or starts after this one ends) is NOT a conflict, so it must
    // stay bookable for any other free date range instead of being treated
    // as fully sold out.
    const overlapping = body.pickup && body.ret
      ? listActiveBookings_().filter(b =>
          b.carId === body.carId && rangesOverlap_(body.pickup, body.ret, b.pickup, b.ret)
        ).length
      : listActiveBookings_().filter(b => b.carId === body.carId).length;
    if (overlapping >= stock) {
      return jsonResponse_({ success: false, error: 'sold_out' });
    }
    const bookingsSheet = getBookingsSheet_();
    bookingsSheet.appendRow([
      body.carId,
      body.carName || '',
      body.customerName || '',
      body.phone || '',
      body.pickup || '',
      body.ret || '',
      body.ref || '',
      new Date(),
      stock,
      body.total || '',
      body.source || 'website',
      body.email || '',
      'none',
      'pending',
      'pending_payment',
    ]);
    // Phone numbers start with "+" (country code, e.g. "+212 6..."), which
    // Google Sheets auto-parses as the start of a formula/number and shows
    // as "#ERROR!" — force that cell to plain text and rewrite it so it
    // displays exactly as entered.
    bookingsSheet.getRange(bookingsSheet.getLastRow(), 4).setNumberFormat('@').setValue(body.phone || '');
    sendBookingEmail_(body);
    if (body.adminPassword) {
      const role = roleForPassword_(body.adminPassword);
      if (role) logActivity_(role, 'manual_booking', (body.source || 'manual') + ' — ' + body.carName);
    }
    return jsonResponse_({ success: true });
  } finally {
    lock.releaseLock();
  }
}

/* Frees up stock automatically once a booking's return date has passed.
   Installed as an hourly time-driven trigger (see v3 setup notes). Also
   checks pending "notify me" requests afterwards, since freeing stock here
   is exactly the moment a sold-out car might become bookable again. */
function autoExpireBookings() {
  const sheet = getBookingsSheet_();
  const rows = sheet.getDataRange().getValues();
  const now = new Date();
  for (let i = rows.length - 1; i >= 1; i--) {
    const ret = rows[i][5];
    if (ret && new Date(ret) < now) {
      sheet.deleteRow(i + 1);
    }
  }
  notifyAvailabilityRequests_();
}

/* For each pending "notify me when available" request, checks whether that
   car currently has free stock; if so, emails the requester (when their
   contact looks like an email — no SMS integration for phone numbers) and
   removes the request so it isn't processed again. Phone-only requests are
   just removed once available, for the team to follow up manually. */
function notifyAvailabilityRequests_() {
  const sheet = getAvailabilityRequestsSheet_();
  const rows = sheet.getDataRange().getValues();
  if (rows.length < 2) return;
  const cars = listCars_();
  const bookedCounts = {};
  listActiveBookings_().forEach(b => { bookedCounts[b.carId] = (bookedCounts[b.carId] || 0) + 1; });
  for (let i = rows.length - 1; i >= 1; i--) {
    const carId = rows[i][0];
    const car = cars.find(c => c.id === carId);
    if (!car) continue;
    const remaining = (car.stock || 1) - (bookedCounts[carId] || 0);
    if (remaining <= 0) continue;
    const contact = rows[i][2];
    if (contact && contact.indexOf('@') !== -1) {
      try {
        MailApp.sendEmail(
          contact,
          'Good news — ' + (rows[i][1] || carId) + ' is available again!',
          'The car you asked about, ' + (rows[i][1] || carId) + ', is now available for booking.\n\nBook it here before it\'s gone: ' + (SITE_URL || '')
        );
      } catch (e) { /* don't fail the whole batch over one email quota/error */ }
    }
    sheet.deleteRow(i + 1);
  }
}

function listActiveBookings_() {
  const rows = getBookingsSheet_().getDataRange().getValues();
  const bookings = [];
  for (let i = 1; i < rows.length; i++) {
    if (!rows[i][0]) continue;
    bookings.push({
      row: i + 1,
      carId: rows[i][0],
      carName: rows[i][1],
      customerName: rows[i][2],
      phone: rows[i][3],
      pickup: rows[i][4],
      ret: rows[i][5],
      ref: rows[i][6],
      bookedAt: rows[i][7],
      stock: rows[i][8],
      total: rows[i][9],
      source: rows[i][10] || 'website',
      email: rows[i][11] || '',
      paymentMethod: rows[i][12] || 'none',
      paymentStatus: rows[i][13] || 'pending',
      bookingStatus: rows[i][14] || 'pending_payment',
    });
  }
  return bookings;
}
/* Finds a booking row by its reference code (unique per booking). Returns
   the 1-indexed sheet row number, or -1 if not found. */
function findBookingRowByRef_(ref) {
  const rows = getBookingsSheet_().getDataRange().getValues();
  for (let i = 1; i < rows.length; i++) {
    if (rows[i][6] === ref) return i + 1;
  }
  return -1;
}

function listSeasons_() {
  const rows = getSeasonsSheet_().getDataRange().getValues();
  const seasons = [];
  for (let i = 1; i < rows.length; i++) {
    if (!rows[i][0]) continue;
    seasons.push({
      row: i + 1,
      name: rows[i][0],
      start: formatDateOnly_(rows[i][1]),
      end: formatDateOnly_(rows[i][2]),
      type: rows[i][3] || 'percent',
      value: rows[i][4],
    });
  }
  return seasons;
}

/* ---- Cars (fleet catalog) ---- */
function listCars_() {
  const rows = getCarsSheet_().getDataRange().getValues();
  const cars = [];
  for (let i = 1; i < rows.length; i++) {
    if (!rows[i][0]) continue;
    cars.push({
      id: rows[i][0],
      name: rows[i][1],
      price: Number(rows[i][2]) || 0,
      seats: Number(rows[i][3]) || 5,
      gearKey: rows[i][4] || 'manual',
      doors: Number(rows[i][5]) || 4,
      fuelKey: rows[i][6] || 'diesel',
      body: rows[i][7] ? String(rows[i][7]).split(',').map(s => s.trim()).filter(Boolean) : [],
      cat: rows[i][8] || 'citadine',
      stock: Number(rows[i][9]) || 1,
      imgSrc: rows[i][10] || '',
      offer: rows[i][11] || (rows[i][2] + '€'),
      sortOrder: rows[i][12] === '' || rows[i][12] == null ? i : Number(rows[i][12]),
    });
  }
  cars.sort((a, b) => a.sortOrder - b.sortOrder);
  return cars;
}

function upsertCar_(car) {
  const sheet = getCarsSheet_();
  const rows = sheet.getDataRange().getValues();
  const rowData = [
    car.id, car.name, Number(car.price) || 0, Number(car.seats) || 5, car.gearKey || 'manual',
    Number(car.doors) || 4, car.fuelKey || 'diesel', Array.isArray(car.body) ? car.body.join(',') : (car.body || ''),
    car.cat || 'citadine', Number(car.stock) || 1, car.imgSrc || '', car.offer || ((Number(car.price) || 0) + '€'),
    car.sortOrder == null ? '' : Number(car.sortOrder),
  ];
  for (let i = 1; i < rows.length; i++) {
    if (rows[i][0] === car.id) {
      sheet.getRange(i + 1, 1, 1, rowData.length).setValues([rowData]);
      return;
    }
  }
  sheet.appendRow(rowData);
}

function deleteCar_(id) {
  const sheet = getCarsSheet_();
  const rows = sheet.getDataRange().getValues();
  for (let i = rows.length - 1; i >= 1; i--) {
    if (rows[i][0] === id) {
      sheet.deleteRow(i + 1);
      return;
    }
  }
}

function duplicateCar_(id) {
  const rows = getCarsSheet_().getDataRange().getValues();
  for (let i = 1; i < rows.length; i++) {
    if (rows[i][0] === id) {
      const newId = id + '-copy-' + Date.now().toString().slice(-6);
      const rowData = rows[i].slice();
      rowData[0] = newId;
      rowData[1] = rowData[1] + ' (Copy)';
      getCarsSheet_().appendRow(rowData);
      return newId;
    }
  }
  return null;
}

/* ---- Settings (key/value store, used for SEO meta overrides) ---- */
function listSettings_() {
  const rows = getSettingsSheet_().getDataRange().getValues();
  const settings = {};
  for (let i = 1; i < rows.length; i++) {
    if (!rows[i][0]) continue;
    settings[rows[i][0]] = rows[i][1];
  }
  return settings;
}
function setSetting_(key, value) {
  const sheet = getSettingsSheet_();
  const rows = sheet.getDataRange().getValues();
  for (let i = 1; i < rows.length; i++) {
    if (rows[i][0] === key) {
      sheet.getRange(i + 1, 2).setValue(value);
      return;
    }
  }
  sheet.appendRow([key, value]);
}

/* ---- Activity log ---- */
function logActivity_(role, action, details) {
  try {
    getActivitySheet_().appendRow([new Date(), role, action, details || '']);
  } catch (e) { /* never let logging break the actual action */ }
}
function listActivityLog_() {
  const rows = getActivitySheet_().getDataRange().getValues();
  const log = [];
  for (let i = Math.max(1, rows.length - 200); i < rows.length; i++) {
    if (!rows[i][0]) continue;
    log.push({ timestamp: rows[i][0], role: rows[i][1], action: rows[i][2], details: rows[i][3] });
  }
  return log.reverse();
}

function formatDateOnly_(value) {
  if (value instanceof Date) {
    return Utilities.formatDate(value, Session.getScriptTimeZone(), 'yyyy-MM-dd');
  }
  return String(value).slice(0, 10);
}

function sendBookingEmail_(body) {
  try {
    const subject = 'New booking: ' + (body.carName || body.carId) + ' — ' + (body.ref || '');
    const lines = [
      'A new booking was just made ' + (body.source && body.source !== 'website' ? '(source: ' + body.source + ')' : 'on the website') + '.',
      '',
      'Reference: ' + (body.ref || ''),
      'Customer: ' + (body.customerName || ''),
      'Phone: ' + (body.phone || ''),
      'Vehicle: ' + (body.carName || body.carId),
      'Pickup: ' + (body.pickup || ''),
      'Return: ' + (body.ret || ''),
      'Total: ' + (body.total || ''),
    ];
    MailApp.sendEmail(NOTIFICATION_EMAIL, subject, lines.join('\n'));
  } catch (e) { /* don't fail the booking over an email quota/error */ }
}

/* Sent to the CUSTOMER (not the agency) once an admin marks a booking's
   payment as "Paid" from the dashboard. Silently does nothing if the
   customer didn't leave an email address (it's an optional field). */
function sendPaymentConfirmedEmail_(row) {
  const email = row[11];
  if (!email) return;
  try {
    const ref = row[6];
    const subject = 'Payment Confirmed - Reservation #' + ref;
    const lines = [
      'Good news — your payment has been verified and your booking is now confirmed!',
      '',
      'Reservation: ' + ref,
      'Vehicle: ' + row[1],
      'Customer: ' + row[2],
      'Pickup: ' + row[4],
      'Return: ' + row[5],
      'Total: ' + row[9],
      '',
      'Thank you for choosing Auto Fluchos Coches.',
    ];
    MailApp.sendEmail(email, subject, lines.join('\n'));
  } catch (e) { /* don't fail the status update over an email quota/error */ }
}
